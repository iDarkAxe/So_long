19void *ft_store_textures(t_mlx *mlx, t_map *map)
20{
21 map->textures.wall.ptr = mlx_xpm_file_to_image(mlx->mlx_ptr,
22 "img/stone_bricks_v2.xpm", &(map->textures.wall.width),
23 &(map->textures.wall.height));
24 if (map->textures.wall.ptr == NULL)
25 return (NULL);
26 map->textures.floor.ptr = mlx_xpm_file_to_image(mlx->mlx_ptr,
27 "img/parquet_versailles.xpm", &(map->textures.floor.width),
28 &(map->textures.floor.height));
29 if (map->textures.floor.ptr == NULL)
30 return (NULL);
31 if (store_textures_exit(mlx, map) == NULL)
32 return (NULL);
33 map->textures.collectible.ptr = mlx_xpm_file_to_image(mlx->mlx_ptr,
34 "img/eye_of_ender.xpm", &(map->textures.collectible.width),
35 &(map->textures.collectible.height));
36 if (map->textures.collectible.ptr == NULL)
37 return (NULL);
38 return (store_textures_player(mlx, map));
39}
