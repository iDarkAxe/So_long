20void free_map(t_map *map, char **map_to_delete, size_t len)
21{
22 size_t i;
23
24 i = 0;
25 if (map == NULL)
26 return ;
27 while (i < (len + 1) && i < (map->height + 1))
28 {
29 if (map_to_delete[i])
30 {
31 free(map_to_delete[i]);
32 map_to_delete[i] = NULL;
33 }
34 i++;
35 }
36 if (map_to_delete != NULL)
37 {
38 free(map_to_delete);
39 map_to_delete = NULL;
40 }
41 free(map);
42 map = NULL;
43}
