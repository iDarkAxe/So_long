28int validate_map(t_map *map)
29{
30 char **new_map;
31 size_t i;
32 int ret;
33
34 i = 0;
35 new_map = malloc(sizeof(char *) * (map->height + 1));
36 if (new_map == NULL)
37 return (-1);
38 while (i <= map->height)
39 {
40 new_map[i] = ft_strdup(map->map[i]);
41 if (new_map[i] == NULL)
42 return ((void)free(new_map), -1);
43 i++;
44 }
45 ret = is_all_accessible(map, new_map);
46 while (i > 0)
47 free(new_map[i-- - 1]);
48 free(new_map);
49 if (ret == 1)
50 return (-1);
51 return (0);
52}

63static int is_wall(size_t x, size_t y, char **map)
64{
65 if (map[y][x] >= 'A' && map[y][x] <= 'Z')
66 {
67 map[y][x] = map[y][x] + 32;
68 return (1);
69 }
70 if (map[y][x] == '0')
71 {
72 map[y][x] = '2';
73 return (1);
74 }
75 return (0);
76}

87static void pathing(t_map *map, size_t x, size_t y, char **new_map)
88{
89 if (is_wall(x + 1, y, new_map))
90 pathing(map, x + 1, y, new_map);
91 if (is_wall(x - 1, y, new_map))
92 pathing(map, x - 1, y, new_map);
93 if (is_wall(x, y + 1, new_map))
94 pathing(map, x, y + 1, new_map);
95 if (is_wall(x, y - 1, new_map))
96 pathing(map, x, y - 1, new_map);
97}

107static int is_all_accessible(t_map *map, char **new_map)
108{
109 t_coordinates map_pos;
110 t_coordinates player_pos;
111
112 map_pos.y = 0;
113 player_pos = find_position(map, 'P');
114 pathing(map, player_pos.x, player_pos.y, new_map);
115 while (map_pos.y < map->height && new_map[map_pos.y])
116 {
117 map_pos.x = 0;
118 while (map_pos.y < map->height && map_pos.x < map->width
119 && new_map[map_pos.y][map_pos.x])
120 {
121 if (new_map[map_pos.y][map_pos.x] == 'E'
122 || new_map[map_pos.y][map_pos.x] == 'C')
123 return (1);
124 map_pos.x++;
125 }
126 map_pos.y++;
127 }
128 return (0);
129}
