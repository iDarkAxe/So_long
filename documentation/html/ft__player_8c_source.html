30void can_player_move(t_mlx *mlx, t_map *map, int direction)
31{
32 t_coordinates old_pos;
33 t_coordinates new_pos;
34
35 old_pos = find_position(map, 'P');
36 map->map[old_pos.y][old_pos.x] = 'p';
37 if (direction == KEY_RIGHT && map->map[old_pos.y][old_pos.x + 1] != '1')
38 new_pos = (t_coordinates){old_pos.x + 1, old_pos.y};
39 else if (direction == KEY_LEFT && map->map[old_pos.y][old_pos.x - 1] != '1')
40 new_pos = (t_coordinates){old_pos.x - 1, old_pos.y};
41 else if (direction == KEY_DOWN && map->map[old_pos.y + 1][old_pos.x] != '1')
42 new_pos = (t_coordinates){old_pos.x, old_pos.y + 1};
43 else if (direction == KEY_UP && map->map[old_pos.y - 1][old_pos.x] != '1')
44 new_pos = (t_coordinates){old_pos.x, old_pos.y - 1};
45 else
46 {
47 map->map[old_pos.y][old_pos.x] = 'P';
48 return ;
49 }
50 if (ft_collectibles_handler(map, mlx, new_pos) == 0)
51 {
52 write(1, "You won!\n", 9);
53 close_window((void *)mlx);
54 }
55 ft_draw_player(mlx, map, direction, old_pos);
56 map->map[old_pos.y][old_pos.x] = '0';
57}
